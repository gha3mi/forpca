   pure subroutine compute_score(this)
      class(tpca), intent(inout) :: this
      integer                    :: i, j

      allocate(this%score(this%nrow,this%ncol))
      do i = 1, this%nrow
         do j = 1, this%npc
            this%score(i, j) = dot_product(this%mean_data(i, :), this%coeff(:, j))
         end do
      end do
   end subroutine compute_score
