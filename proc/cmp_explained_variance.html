   pure subroutine cmp_explained_variance(this)
      class(tpca), intent(inout) :: this
      real(rk)                   :: sum_latent

      sum_latent = sum(this%latent(1:this%npc))
      this%explained_variance = this%latent / sum_latent
   end subroutine cmp_explained_variance
